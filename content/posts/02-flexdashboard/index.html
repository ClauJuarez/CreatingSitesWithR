PIB per cápita por continente

data <- gapminder %>% filter(year == 2007)

ggplot(data, aes(continent, gdpPercap, fill = continent)) +
  geom_col() +
  theme_minimal() +
  labs(
    title = "PIB per cápita por continente (2007)",
    x = "Continente", y = "PIB per cápita"
  )
